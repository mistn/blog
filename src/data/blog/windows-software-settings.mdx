export const AppCard = ({ name, desc, icon, link }) => {
  let iconUrl = icon;
  if (!iconUrl && link) {
    try {
      const domain = new URL(link).hostname;
      const svgDomains = ['github.com', 'pandoc.org', 'ditto-cp.sourceforge.io'];
      const localPath = svgDomains.includes(domain) ? `/software-icons/${domain}.svg` : `/software-icons/${domain}.png`;
      const apiFallback = `https://icons.duckduckgo.com/ip3/${domain}.ico`;
      iconUrl = localPath;
    } catch (e) {
      iconUrl = '';
    }
  }

  return (
    <a href={link} target="_blank" rel="noopener noreferrer" class="flex items-center gap-3 py-2.5 group no-underline text-inherit">
      {iconUrl ? (
        <img
          src={iconUrl}
          alt={name}
          class={`w-8 h-8 shrink-0 rounded-md !border-none !shadow-none !bg-transparent !p-0 !m-0 group-hover:scale-105 transition-transform duration-300 object-contain${iconUrl.endsWith('.svg') ? ' dark:brightness-0 dark:invert' : ''}`}
          loading="lazy"
          onerror="this.onerror=null;if(!this.dataset.fallback){this.dataset.fallback='1';var d=this.closest('a').href;var u=new URL(d);this.src='https://icons.duckduckgo.com/ip3/'+u.hostname+'.ico'}else{this.src='/software-icons/_fallback.svg'}"
        />
      ) : (
        <img src="/software-icons/_fallback.svg" alt={name} class="w-8 h-8 shrink-0 rounded-md !border-none !shadow-none !bg-transparent !p-0 !m-0 dark:brightness-0 dark:invert object-contain" loading="lazy" />
      )}
      <div class="flex-1 min-w-0">
        <div class="font-medium text-gray-900 dark:text-gray-100 group-hover:text-blue-600 dark:group-hover:text-blue-400 transition-colors text-[14px] leading-tight">
          {name}
        </div>
        <div class="text-[13px] text-gray-500 dark:text-gray-400 mt-1">
          {desc}
        </div>
      </div>
    </a>
  );
};
